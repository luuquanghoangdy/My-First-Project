--cau 1 : 
	select KHACHHANG.* from KHACHHANG join HOPDONGCHOTHUE on KHACHHANG.MaKH=HOPDONGCHOTHUE.MaKH join CHITIETHOPDONG on HOPDONGCHOTHUE.MaHopDong=CHITIETHOPDONG.MaHopDong join LOAIDICHVU on CHITIETHOPDONG.MaLoaiDV=LOAIDICHVU.MaLoaiDV where LOAIDICHVU.MoTaLoaiDV='Chung Tet'

	--Cau 2:

	select CHITIETHOPDONG.* from CHITIETHOPDONG join CAYCANH on CHITIETHOPDONG.MaCayCanh= CAYCANH.MaCayCanh join HOPDONGCHOTHUE on CHITIETHOPDONG.MaHopDong=HOPDONGCHOTHUE.MaHopDong join KHACHHANG on KHACHHANG.MaKH=HOPDONGCHOTHUE.MaKH where CAYCANH.DonGiaChoThue<50000 and KHACHHANG.DiaChiKH='Hai Chau'

	--Cau 3:
	select sum (SoTienDuocGiamGia) from HOPDONGCHOTHUE

	--Cau 4:
	select KHACHHANG.MaKH,KHACHHANG.HoTenKH, DATEDIFF(day,CHITIETHOPDONG.NgayBatDauChoThue,CHITIETHOPDONG.NgayKetThucChoThue) as SoNgayThue from KHACHHANG join HOPDONGCHOTHUE on HOPDONGCHOTHUE.MaKH=KHACHHANG.MaKH join CHITIETHOPDONG on CHITIETHOPDONG.MaHopDong=HOPDONGCHOTHUE.MaHopDong group by KHACHHANG.MaKH,KHACHHANG.HoTenKH,CHITIETHOPDONG.NgayBatDauChoThue,CHITIETHOPDONG.NgayKetThucChoThue

	--Cau 5:
	select top(1)CAYCANH.MaLoaiCC , count (CAYCANH.MaLoaiCC)as SoLanThue  from CAYCANH join CHITIETHOPDONG on CHITIETHOPDONG.MaCayCanh= CAYCANH.MaCayCanh group by CAYCANH.MaLoaiCC order by SoLanThue DESC

	--Cau 6 :

	select KHACHHANG.MaKH, count (HOPDONGCHOTHUE.MaKH) SoLanThue from KHACHHANG left outer join  HOPDONGCHOTHUE on KHACHHANG.MaKH=HOPDONGCHOTHUE.MaKH 
	group by KHACHHANG.MaKH,HOPDONGCHOTHUE.MaKH

	--Cau 7 :
	select top(2) KHACHHANG.MaKH, count (HOPDONGCHOTHUE.MaKH) SoLanThue from KHACHHANG join  HOPDONGCHOTHUE on KHACHHANG.MaKH=HOPDONGCHOTHUE.MaKH 
	group by KHACHHANG.MaKH,HOPDONGCHOTHUE.MaKH order by SoLanThue ASC
	--Cau 8 : 
	select LOAIDICHVU.MaLoaiDV,count (CHITIETHOPDONG.MaCayCanh) SoLanThue,CAYCANH.MaCayCanh from LOAIDICHVU join CHITIETHOPDONG on LOAIDICHVU.MaLoaiDV=CHITIETHOPDONG.MaLoaiDV join CAYCANH on CAYCANH.MaCayCanh=CHITIETHOPDONG.MaCayCanh group by CAYCANH.MaCayCanh,LOAIDICHVU.MaLoaiDV
	--Cau 10 :
	select count(MaKH) as SoKhachHang from KHACHHANG where MaKH in(select distinct KHACHHANG.MaKH from KHACHHANG join HOPDONGCHOTHUE on HOPDONGCHOTHUE.MaKH=KHACHHANG.MaKH join CHITIETHOPDONG on HOPDONGCHOTHUE.MaHopDong=CHITIETHOPDONG.MaHopDong where CHITIETHOPDONG.SoLuong>100)
